125 void
126 ResultEncoder::put(const char* term, const size_t term_len)
127 {
128  /* Len of the whole buffer after coping */
129  const size_t new_len = term_len + m_result_len;
130  const bool allocated = m_first_segment != NULL;
131  const bool large = allocated
132  || m_default_buf == NULL
133  || new_len > m_default_len;
134  const bool no_alloc = allocated && term_len <= m_left_len;
135 
136  if (!large || no_alloc)
137  {
138  /* Whole term can be stored in the preallocated area. */
139  char* dest = m_current_buf;
140  const char* src = term;
141  memcpy(dest, src, term_len);
142  m_current_buf += term_len;
143  m_left_len -= term_len;
144  }
145  else
146  {
147  /* part1 will stay in a default buffer. */
148  const size_t part1_len = m_left_len;
149  const size_t part2_len = term_len - part1_len;
150  const size_t new_segment_len = part2_len + RESERVED_LEN;
151 
152  /* Create new data segment. */
153  DataSegment*
154  new_segment = this->alloc(new_segment_len);
155  new_segment->size = new_segment_len;
156  new_segment->next = NULL;
157 
158  const char* part1_src = term;
159  const char* part2_src = term + part1_len;
160 
161  char* part1_dest = m_current_buf;
162  char* part2_dest = new_segment->data;
163 
164  /* Copy data */
165  memcpy(part1_dest, part1_src, part1_len);
166  memcpy(part2_dest, part2_src, part2_len);
167 
168  if (m_first_segment == NULL) {
169  /* Set as a first segment. */
170  m_first_segment = new_segment;
171  } else {
172  /* Link to a last segment. */
173  m_last_segment->next = new_segment;
174  }
175 
176  /* Save state */
177  m_last_segment = new_segment;
178  m_current_buf = part2_dest + part2_len;
179  m_left_len = RESERVED_LEN;
180  }
181 
182  m_result_len = new_len;
183 }

190 ResultEncoder::clear() {
191  if (m_result_len > m_default_len)
192  {
193  do
194  {
195  assert(m_first_segment != NULL);
196 
197  /* No infinite cycles. */
198  assert(m_first_segment->next != m_first_segment);
199 
200  /* Use last_segment as a temp variable. */

202 
203  /* Move a pointer on the next segment. */
204  m_first_segment = m_first_segment->next;
